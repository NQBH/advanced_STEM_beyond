using namespace std;

int main() {
    float a, b, c;
    cin >> a >> b >> c;

    float max = a, min = a;

    if (b > max) max = b;
    if (c > max) max = c;

    if (b < min) min = b;
    if (c < min) min = c;

    cout << "Max = " << max << endl;
    cout << "Min = " << min << endl;

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int duong = 0, am = 0, bang0 = 0;
    float x;

    for (int i = 1; i <= n; i++) {
        cin >> x;
        if (x > 0) duong++;
        else if (x < 0) am++;
        else bang0++;
    }

    cout << "So duong: " << duong << endl;
    cout << "So am: " << am << endl;
    cout << "So bang 0: " << bang0 << endl;

    return 0;
}
