#include <iostream>
using namespace std;

int main() {
    float a[3];
    for (int i = 0; i < 3; i++)
        cin >> a[i];

    // sắp xếp tăng dần
    for (int i = 0; i < 2; i++)
        for (int j = i + 1; j < 3; j++)
            if (a[i] > a[j]) {
                float t = a[i];
                a[i] = a[j];
                a[j] = t;
            }

    cout << "Min = " << a[0] << endl;
    cout << "Max = " << a[2] << endl;
    return 0;
}

#include <iostream>
using namespace std;

void dem(float x, int &duong, int &am, int &bang0) {
    if (x > 0) duong++;
    else if (x < 0) am++;
    else bang0++;
}

int main() {
    int n;
    cin >> n;

    int duong = 0, am = 0, bang0 = 0;
    float x;

    for (int i = 0; i < n; i++) {
        cin >> x;
        dem(x, duong, am, bang0);
    }

    cout << "Duong: " << duong << endl;
    cout << "Am: " << am << endl;
    cout << "Bang 0: " << bang0 << endl;
    return 0;
}

int main() {
    int n, x;
    cin >> n;

    int chan = 0, le = 0;

    for (int i = 0; i < n; i++) {
        cin >> x;
        if ((x & 1) == 0)
            chan++;
        else
            le++;
    }

    cout << "Chan: " << chan << endl;
