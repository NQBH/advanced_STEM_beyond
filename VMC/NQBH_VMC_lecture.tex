\section{Algebra -- Đại Số}
Gồm 2 phần: Đại số tuyến tính (Linear Algebra) \& Đại số trừu tượng (Abstract Algebra). Olympic Toán Sinh Viên thường nghiêng về Đại số tuyến tính nhiều hơn.

\paragraph{Notation -- Ký hiệu.} $\mathbb{N},\mathbb{N}^\star,\mathbb{Z},\mathbb{Q},\mathbb{R}$ lần lượt là các tập số tự nhiên, số nguyên dương, số nguyên, số hữu tỷ, số thực, $\overline{\mathbb{R}} = \mathbb{R}\cup\{\pm\infty\}$ là tập số thực mở rộng. Với $A$ là 1 trong các tập số này thì $A_{\ge a} = \{x\in A;x\ge a\}$, các tập $A_{> a},A_{\le a},A_{< a}$ được định nghĩa tương tự.

Ta sử dụng ký hiệu $F$ để chỉ 1 {\it trường} (field). Với sinh viên các ngành không phải Toán \& cận Toán thì chỉ cần quan tâm trường hợp $F = \mathbb{R}$, còn các ngành cận Toán như Khoa học Máy tính, Công nghệ thông tin, Vật lý, etc., thì chỉ cần quan tâm các trường hợp $F\in\{\mathbb{R},\mathbb{C}\}$, với sinh viên các ngành Toán, Toán--Tin, Toán--Cơ--Tin, nên quan tâm $F\in\{\mathbb{R},\mathbb{C},\mathbb{Z}_p\mbox{ với } p \mbox{ nguyên tố}\}$ \& trường hữu hạn gồm $q$ phần tử $F_q$, vì $\mathbb{Z}_p$ với $p$ nguyên tố, để làm các bài Đa thức \& Số học trong đề Đại số.

Với 1 ma trận $A\in M_{m\times n}(F)$, gọi ${\bf r}_i(A),{\bf c}_j(A)$ lần lượt là dòng (row) thứ $i\in[m]$, cột (column) thứ $j\in[n]$ của ma trận $A$:
\begin{align*}
	{\bf r}_i(A)&\coloneqq(a_{i1},a_{i2},\ldots,a_{in})\in M_{1\times n}(F),\ \forall i\in[m],\\
	{\bf c}_j(A)&\coloneqq(a_{1j},a_{2j},\ldots,a_{mj})^\top = \begin{pmatrix}
		a_{1j}\\a_{2j}\\\vdots\\a_{mj}
	\end{pmatrix}\in M_{m\times1}(F) = F^m,\ \forall j\in[n].
\end{align*}

\subsection{Vector space -- Không gian vector}

\paragraph{Resources.} \cite[Chương 1]{Hoa_linear_algebra}, \cite{Hung_linear_algebra}, \cite{Trung_linear_algebra}.

\begin{dinhnghia}[Không gian vector]
	\label{def: vector space}
	Tập hợp $V\ne\emptyset$ cùng với {\rm phép cộng vector} $+:V\times V\to V:(x,y)\mapsto x + y$ \& {\rm phép nhân vô hướng} $\cdot:F\times V\to V:(\alpha,x)\mapsto\alpha x$ được gọi là {\rm không gian vector} trên trường $F$ nếu các điều kiện sau thỏa mãn:
	\item(i) Tính chất kết hợp của phép cộng vector: $(x + y) + z = x + (y + z)$, $\forall x,y,z\in V$.
	\item(ii) Tính chất giao hoán của phép cộng vector: $x + y = y + x$, $\forall x,y,z\in V$.
	\item(iii) Tồn tại vector $0_V$, gọi là {\rm vector không}, đóng vai trò là phần tử trung hòa của phép cộng vector trên $V$, có tính chất $0_V + x = x + 0_V = x$, $\forall x\in V$.
	\item(iv) Tồn tại vector $-x\in V$, gọi là {\rm vector đối} của vector $x\in V$, sao cho $x + (-x) = (-x) + x = 0_V$, $\forall x\in V$.
	\item(v) $(\alpha\beta)x = \alpha(\beta x)$, $\forall\alpha,\beta\in F$, $\forall x\in V$.
	\item(vi) $(\alpha + \beta)x = \alpha x + \beta x$, $\forall\alpha,\beta\in F$, $\forall x\in V$.
	\item(vii) $\alpha(x + y) = \alpha x + \alpha y$, $\forall\alpha\in F$, $\forall x,y\in V$.
	\item(viii) $1x = x$, $\forall x\in V$.\footnote{$x1$ không có nghĩa vì ta chỉ mới định nghĩa phép toán nhân vô hướng với vector, còn phép toán nhân vector với vô hướng chưa được định nghĩa: thứ tự các biến của 1 phép toán rất quan trọng.}
	
	Ta gọi phần tử của $V$ là {\rm vector}, phần tử của $K$ là {\rm phần tử vô hướng}.
\end{dinhnghia}

\begin{bode}[\cite{Hoa_linear_algebra}, Bổ đề 1.2, p. 16]
	Cho $U$  là 1 tập con khác rỗng của không gian vector $V$ trên trường $F$. 2 điều kiện sau là tương đương:
	\item(i) $U$ là không gian con của $V$, i.e., $U$ đóng với phép cộng vector \& phép nhân vô hướng, \& cùng với 2 phép toán cảm sinh, bản thân nó là 1 không gian vector trên trường $F$.
	\item(ii) $U$ đóng với phép cộng vector \& phép nhân vô hướng.
\end{bode}

\begin{hequa}[\cite{Hoa_linear_algebra}, Hệ quả 1.3, p. 16]
	Nếu $U$ là không gian con của không gian vector $V$, thì $0_V\in U$.
\end{hequa}
Để chứng minh 1 tập hợp $X$ là 1 không gian vector (trên trường $F$), có 2 phương pháp chính:
\begin{enumerate}
	\item Sử dụng định nghĩa \ref{def: vector space} của không gian vector.
	\item Kiểm tra xem $X$ có là không gian con của 1 không gian vector $V$ nào đó không, nhờ vào \cite[Bổ đề 1.2, p. 16]{Hoa_linear_algebra}.
\end{enumerate}

\paragraph{Exercises.} Đọc hiểu các ví dụ \cite[Chương 1]{Hoa_linear_algebra} \& làm \cite[Bài 1.1--1.18]{Hoa_linear_algebra}.

\subsubsection{Linear independence \& spanning set -- Độc lập tuyến tính \& hệ sinh}

\begin{dinhnghia}[Phụ thuộc tuyến tính, độc lập tuyến tính, tổ hợp tuyến tính, \cite{Hoa_linear_algebra}, ĐN2.1, p. 20]
	Cho $V$ là 1 không gian vector trên trường $F$. Các vector $\{v_i\}_{i=1}^n$ được gọi là {\rm phụ thuộc tuyến tính} nếu tồn tại $\alpha_1,\ldots,\alpha_n\in F$ không đồng thời bằng $0_F$ (i.e., $(\alpha_1,\ldots,\alpha_n)\ne(0_F,\ldots,0_F)\in F^n$) sao cho
	\begin{equation*}
		\sum_{i=1}^n \alpha_iv_i = \alpha_1v_1 + \cdots + \alpha_nv_n = 0_V.
	\end{equation*}
	Tập vector $S$ được gọi là {\rm phụ thuộc tuyến tính} nếu nó chứa 1 hệ hữu hạn vector phụ thuộc tuyến tính.
	
	Tập vector không phụ thuộc tuyến tính được gọi là {\rm độc lập tuyến tính}.
	
	Biểu thức $\sum_{i=1}^n \alpha_iv_i = \alpha_1v_1 + \cdots + \alpha_nv_n$ được gọi là 1 {\rm tổ hợp tuyến tính} của các phần tử $v_1,\ldots,v_n$. Nếu $v$ là 1 tổ hợp tuyến tính của $v_1,\ldots,v_n$ thì ta cũng nói $v$ {\rm biểu diễn tuyến tính} qua $v_1,\ldots,v_n$.
\end{dinhnghia}

\begin{hequa}[\cite{Hoa_linear_algebra}, HQ2.2, p. 22]
	(i) Tập con của 1 tập độc lập tuyến tính là độc lập tuyến tính.
	\item(ii) 1 tập chứa 1 tập phụ thuộc tuyến tính là phụ thuộc tuyến tính.
\end{hequa}

\begin{bode}[\cite{Hoa_linear_algebra}, BĐ2.3, p. 22]
	Tập $S$ phụ thuộc tuyến tính khi nó chứa 1 vector biểu diễn tuyến tính qua các vector còn lại.
\end{bode}

\begin{hequa}[\cite{Hoa_linear_algebra}, HQ2.4, p. 22]
	Nếu $\dim V = n$ thì mọi tập từ $n + 1$ phần tử trở lên đều phụ thuộc tuyến tính, còn tập có tối đa $n - 1$ phần tử không thể là hệ sinh.
\end{hequa}
{\bf Intuition.} Hiểu theo kiểu tập có $\le n - 1$ phần tử thì không đủ khả năng mô tả (expressivity) hay biểu diễn tuyến tính của tất cả các vector trong không gian vector $V$ $n$ chiều, còn tập có $n + 1$ phần tử thì theo lại bị bão hòa nên sẽ phải có phần tử biểu diễn tuyến tính được qua các phần tử còn lại.

\begin{dinhly}[\cite{Hoa_linear_algebra}, ĐL2.5, p. 22]
	1 ma trận vuông có định thức khác $0$ $\Leftrightarrow$ các vector dòng (tương ứng các vector cột) của nó độc lập tuyến tính.
\end{dinhly}

\begin{dinhnghia}[\cite{Hoa_linear_algebra}, ĐN2.6, p. 23]
	Cho $S$ là 1 tập con của không gian vector $V$. Ta gọi tập hợp các tổ hợp tuyến tính của các phần tử thuộc $S$ là {\rm bao tuyến tính} của $S$ \& ký hiệu là $E(S)$ hay $\operatorname{span}(S)$. $S$ được gọi là {\rm hệ sinh} của $V$ nếu $E(S) = V$. Ta nói hệ sinh $S$ là {\rm tối tiểu} nếu $S$ không chứa 1 tập con thực sự cũng là hệ sinh.
\end{dinhnghia}
{\bf Intuition.} Hệ sinh có thể hiểu theo kiểu là khung xương sống của 1 không gian vector.

\begin{bode}[\cite{Hoa_linear_algebra}, BĐ2.7, p. 23]
	$E(S)$ là không gian con nhỏ nhất của $V$ chứa $S$.
\end{bode}

\subsubsection{Cơ sở, chiều, \& hạng của 1 hệ vector}

\begin{dinhnghia}[\cite{Hoa_linear_algebra}, ĐN3.1, p. 29]
	Hệ vector $S\subset V$ được gọi là {\rm cơ sở (base)}  của $V$ nếu mỗi vector của $V$ có thể được biểu diễn tuyến tính duy nhất qua hệ $S$.
	
	Nếu tập được sắp hoàn toàn $S = \{v_i\}_{i\in I}$là cơ sở của $V$ \& $v\in V$, thì bộ phần tử $\{\alpha_i\}_{i\in I}$ được gọi là {\rm tọa độ (coordinate)} của $v$ theo $S$ khi $v = \sum_{i\in I} \alpha_iv_i$.
\end{dinhnghia}

\begin{dinhly}[Định lý đánh tráo Steinitz, \cite{Hoa_linear_algebra}, ĐL3.2, p. 29]
	Số vector trong mọi cơ sở của không gian vector $V$ hữu hạn sinh là như nhau. Số này được gọi là chiều của $V$, \& được ký hiệu là $\dim V$.
\end{dinhly}
Khi cố định 1 cở sở $B = \{b_1,\ldots,b_n\}\subset V$ với $\dim V = n$, thì mỗi vector $v\in V$ được xác định duy nhất bằng tọa độ của nó trong $B$, thường được ký hiệu bởi $[v]_B\coloneqq(\alpha_1,\ldots,\alpha_n)^\top$ (1 vector cột $n$ chiều) với $v = \sum_{i=1}^n \alpha_ib_i$.

\subsection{Some basic properties of matrices -- Vài tính chất cơ bản của ma trận}

\begin{remark}
	Ma trận nói chung không có tính chất giao hoán, i.e., $AB$ có thể khác $BA$, nhưng có tính chất kết hợp, i.e., $(AB)C = A(BC)$.
\end{remark}

\begin{remark}
	Khi 2 ma trận $A,B$ giao hoán với nhau, i.e., $AB = BA$ thì các hằng đẳng thức (đại số) đáng nhớ sẽ áp dụng được cho $A,B$, e.g.,
	\begin{align*}
		(A + B)^2 &= (A + B)(A + B) = A^2 + AB + BA + B^2 = A^2 + AB + AB + B^2 = A^2 + 2AB + B^2,\\
		(A - B)^2 &= (A - B)(A - B) = A^2 - AB - BA + B^2 = A^2 - AB - AB + B^2 = A^2 - 2AB + B^2. 
	\end{align*}
	Chú ý đẳng thức $AB = BA$ đã hàm ý luôn việc $A,B\in M_n(\mathbb{R})$ (i.e., 2 ma trận $A,B$ phải vuông \& cùng cấp). Thật vậy, giả sử $A\in  M_{m\times n}(F),B\in M_{p\times q}(F)$, để tích $AB$ xác định thì $n = p$, \& để tích $BA$ xác định thì $q = m$. Khi đó $AB\in M_m(F),BA\in M_n(F)$, đẳng thức $AB = BA$ suy ra $m = n$. Vậy $A,B\in M_n(\mathbb{R})$.
\end{remark}
Tương tự, ta có kết quả sau:

\begin{bode}[Các hằng đẳng thức cho 2 ma trận giao hoán]
	Cho 2 ma trận $A,B\in M_n(\mathbb{C})$ giao hoán, i.e., $AB = BA$, \& 1 hằng đẳng thức đại số 2 biến có dạng $f(x,y) = 0$. Khi đó $f(A,B) = 0_n$.
	
	Nói riêng, ta có công thức nhị thức Newton sau cho 2 ma trận $A,B$:
	\begin{align*}
		(A + B)^n &= \sum_{i=0}^n C_n^iA^{n - i}B^i,\ \forall n\in\mathbb{N}^\star,\\
		(A - B)^n &= \sum_{i=0}^n (-1)^iC_n^iA^{n - i}B^i,\ \forall n\in\mathbb{N}^\star,\\
		A^n - B^n &= (A - B)\sum_{i=0}^{n-1} A^{n - 1 - i}B^i,\ \forall n\in\mathbb{N}^\star\\
		A^{2n + 1} - B^{2n + 1} &= (A + B)\sum_{i=0}^{2n} (-1)^iA^{2n - i}B^i,\ \forall n\in\mathbb{N}^\star
	\end{align*}
\end{bode}

\subsection{Rank \& determinant of a matrix -- Hạng \& định thức của ma trận}

\begin{tip}
	Nếu bài Đại số cho 1 ma trận $4\times4$, thường là có tham số $a$ hoặc $m$, cố gắng tìm cấu trúc đặc biệt của ma trận đó, e.g., tổng các hàng{\tt/}cột, tổ hợp tuyến tính dễ thấy của các hàng{\tt/}cột.
\end{tip}
E.g.:

\begin{baitoan}[VMC2025A, Đại số, 1.]
	Cho $a\in\mathbb{R}$,
	\begin{equation*}
		M(a) = \begin{pmatrix}
			a & 2023 & 2024 & 2025\\2025 & 2024 & 2023 & a\\2024 & 2025 & a & 2023\\2023 & a & 2025 & 2024
		\end{pmatrix}
	\end{equation*}
	\item(a) Tính $\operatorname{rank}M(2022)$.
	\item(b) Tính $\det M(a)$ theo $a$.
	\item(c) Xét hệ phương trình tuyến tính thuần nhất $M(a)X = {\bf0}_4$, trong đó $X = (x_1,x_2,x_3,x_4)^\top$ là vector cột gồm 4 tọa độ $x_1,x_2,x_3,x_4\in\mathbb{R}$. Tìm số chiều của không gian nghiệm của hệ phương trình tuyến tính thuần nhất $M(2022)X = {\bf0}_4$. Khi nào hệ phương trình tuyến tính thuần nhất $M(a)X = {\bf0}_4$ chỉ có hữu hạn nghiệm?
\end{baitoan}

\begin{proof}[Phân tích]
	Đập ngay vào mắt, $M(a)$ có mỗi dòng \& mỗi cột đều chứa đúng 4 số $a,2023,2024,2025$ nên nếu thực hiện phép biến đổi sơ cấp 3 trên dòng{\tt/}cột bằng cách gán tổng của 4 dòng{\tt/}cột cho 1 dòng, e.g., dòng cuối $r_4$, thì sẽ thu được:
	\begin{equation*}
		{\bf r}_4(M(a))\coloneqq\sum_{i=1}^4 {\bf r}_i(M(a)) = (a + 2023 + 2024 + 2025)(1,1,1,1),
	\end{equation*}
	nên nếu $a = -(2023 + 2024 + 2025)$ thì dòng $r_4$ này bằng $(0,0,0,0)$, nên đa thức đặc trưng $p_A(x)$ sẽ có 1 nghiệm là $x = -(2023 + 2024 + 2025) = -6072$.
\end{proof}

\subsection{Diagonalizability of matrices -- Tính chéo hóa được của ma trận}

\begin{dinhly}[Tính chéo hóa của ma trận]
	Nếu ma trận vuông $A\in M_n(\mathbb{R})$ có $n$ giá trị riêng phân biệt đôi một thì $A$ chéo hóa được.
\end{dinhly}

\begin{dinhly}[\cite{Hoa_linear_algebra}, Thm. 18.3]
	Cho $\{v_i\}_{i=1}^r$ là các vector riêng của $\varphi$ ứng với $r$ giá trị riêng $\{\lambda_i\}_{i=1}^r$ đôi một khác nhau. Khi đó $\{v_i\}_{i=1}^r$ độc lập tuyến tính.
\end{dinhly}

\begin{dinhnghia}
	1 toán tử tuyến tính được gọi là {\rm chéo hóa được} nếu nó có 1 ma trận biểu diễn là ma trận đường chéo. 1 ma trận vuông $A$ được gọi là {\rm chéo hóa được} nếu $A$ đồng dạng với ma trận đường chéo, i.e., tồn tại ma trận khả nghịch $P$ để $P^{-1}AP$ là 1 ma trận đường chéo.
\end{dinhnghia}

\begin{dinhly}[\cite{Hoa_linear_algebra}, Thm. 18.4]
	Cho $\varphi$ là 1 toán tử tuyến tính của không gian vector $V$ chiều $n$. Khi đó
	\item(i) $\varphi$ chéo hóa được khi \& chỉ khi $V$ có 1 cơ sở gồm các vector riêng.
	\item(ii) Nếu $\varphi$ có $n$ giá trị riêng khác nhau thì $\varphi$ chéo hóa được.
\end{dinhly}

\begin{baitoan}[VMC2025ĐSA2B4, \cite{VMS_VMC2025}, p. 21]
	Giả sử ban đầu học sinh các trường phổ thông ở Quận 1 chỉ mua bánh mì của Công ty A. Sau đó Công ty B cũng tham gia vào việc cung cấp bánh mì cho thị trường này. Nghiên cứu cho thấy sau mỗi quý (3 tháng) có $30\%$ khách hàng của Công ty A chuyển sang dùng sản phẩm của Công ty B, số còn lại vẫn dùng của Công ty A. Đồng thời có $40\%$ khách hàng của Công ty B chuyển sang dùng của Công ty A, số còn lại vẫn dùng của Công ty B.
	\item(a) Sau $18$ tháng kể từ khi Công ty B tham gia vào thị trường, tính tỷ lệ khách hàng (trên tổng số học sinh của toàn Quận 1) mua bánh mỳ của mỗi công ty tương ứng.
	\item(b) Giám đốc Công ty A quyết định sẽ ngừng kinh doanh mặt hàng bánh mì nếu tỷ lệ khách hàng của họ (trên tổng số) ít hơn $50\%$. Hỏi với số liệu như trên, Công ty A có thời điểm nào cần quyết định ngừng kinh doanh mặt hàng này không? Tại sao?
\end{baitoan}

\begin{proof}[Giải]
	Gọi $a_n,b_n$ lần lượt là tỷ lệ khách hàng trên tổng số ở chu kỳ thứ $n\in\mathbb{N}$, với $a_0 = 1,b_0 = 0$ (ban đầu chỉ có công ty A bán, công ty B chưa tham gia), lập được hệ phương trình truy hồi tuyến tính:
	\begin{equation*}
		\left\{\begin{split}
			a_{n+1} &= 0.7a_n + 0.4b_n,\\
			b_{n+1} &= 0.3a_n + 0.6b_n.
		\end{split}\right.
	\end{equation*}
	Đặt
	\begin{equation*}
		A = \begin{pmatrix}
			0.7 & 0.4\\0.3 & 0.6
		\end{pmatrix},
	\end{equation*}
	thì hệ phương trình truy hồi tuyến tính trên trở thành
	\begin{equation*}
		\begin{pmatrix}
			a_{n+1}\\b_{n+1}
		\end{pmatrix} = A\begin{pmatrix}
			a_n\\b_n
		\end{pmatrix}.
	\end{equation*}
	Chéo hóa ma trận $A$:
	\begin{equation*}
		A = \begin{pmatrix}
			\frac{4}{3} & -1\\1 & 1
		\end{pmatrix}\begin{pmatrix}
			1 & 0\\0 & \frac{3}{10}
		\end{pmatrix}\begin{pmatrix}
			\frac{4}{3} & -1\\1 & 1
		\end{pmatrix}^{-1}.
	\end{equation*}
	Lặp bằng hệ phương trình truy hồi tuyến tính (hoặc dễ dàng chứng minh bằng quy nạp):
	\begin{equation*}
		\begin{pmatrix}
			a_n\\b_n
		\end{pmatrix} = A\begin{pmatrix}
			a_{n-1}\\b_{n-1}
		\end{pmatrix} = A^2\begin{pmatrix}
			a_{n-2}\\b_{n-2}
		\end{pmatrix} = \cdots = A^n\begin{pmatrix}
			a_0\\b_0
		\end{pmatrix} = \begin{pmatrix}
			\frac{4}{3} & -1\\1 & 1
		\end{pmatrix}\begin{pmatrix}
			1 & 0\\0 & \left(\frac{3}{10}\right)^n
		\end{pmatrix}\begin{pmatrix}
			\frac{4}{3} & -1\\1 & 1
		\end{pmatrix}^{-1}\begin{pmatrix}
			1\\0
		\end{pmatrix} = \begin{pmatrix}
			\frac{4}{7} + \frac{3}{7}\left(\frac{3}{10}\right)^n\\\frac{3}{7} - \frac{3}{7}\left(\frac{3}{10}\right)^n
		\end{pmatrix},\ \forall n\in\mathbb{N}.
	\end{equation*}
	Sau 18 tháng $= 6$ quý, tỷ lệ khách hàng sử dụng sản phẩm 2 Công ty A, B tương ứng là:
	\begin{equation*}
		\begin{pmatrix}
			a_6\\b_6
		\end{pmatrix} = \begin{pmatrix}
			\frac{4}{7} + \frac{3}{7}\left(\frac{3}{10}\right)^6\\\frac{3}{7} - \frac{3}{7}\left(\frac{3}{10}\right)^6
		\end{pmatrix} = \begin{pmatrix}
			\frac{4002187}{7000000}\vspace{1mm}\\\frac{2997813}{7000000}.
		\end{pmatrix}
	\end{equation*}
	\item(b) Có dãy $a_n = \frac{4}{7} + \frac{3}{7}\left(\frac{3}{10}\right)^n$ giảm \& $\lim_{n\to\infty} a_n = \frac{4}{7} > 0.5$, nên tỷ lệ khách hàng sử dụng sản phẩm của Công ty A luôn $> \frac{4}{7} > \frac{3.5}{7} = 0.5$, nên Công ty A không cần phải thay đổi mặt hàng.
\end{proof}

\begin{remark}
	Bài toán trên là 1 trường hợp cụ thể của bài toán chuyển đổi trạng thái thông qua 1 hệ phương trình truy hồi tuyến tính được mô tả như sau: Giả sử có $n\in\mathbb{N}_{\ge2}$ đối tượng $\{A_i\}_{i=1}^n$ (agents, ở bài toán trên là 2 công ty A, B), mỗi đối tượng $A_i$ quan tâm đến 1 đại lượng $f_i(t_j)$ tại mỗi thời điểm $t_j\in[0,\infty)$, $\forall i\in[n]$, $\forall j\in\mathbb{N}$. Trạng thái ban đầu được cho trước: $(f_1(t_0),\ldots,f_n(t_0)) = (\alpha_1,\ldots,\alpha_n)$, \& quá trình chuyển trạng thái từ mốc thời gian $t_n$ sang $t_{n+1}$ được mô tả bởi hệ phương trình truy hồi tuyến tính sau:
	\begin{equation*}
		f_i(t_{n+1}) = \sum_{j=1}^n a_{ij}f_j(t_n),\ \forall i\in[n],\ \forall n\in\mathbb{N},
	\end{equation*}
	viết cụ thể:
	\begin{equation*}
		\left\{\begin{split}
			f_1(t_{n+1}) &= \sum_{j=1}^n a_{1j}f_j(t_n) = a_{11}f_1(t_n) + a_{12}f_2(t_n) + \cdots + a_{1n}f_n(t_n),\\
			f_2(t_{n+1}) &= \sum_{j=1}^n a_{2j}f_j(t_n) = a_{21}f_1(t_n) + a_{22}f_2(t_n) + \cdots + a_{2n}f_n(t_n),\\
			&\ldots\\
			f_n(t_{n+1}) &= \sum_{j=1}^n a_{nj}f_j(t_n) = a_{n1}f_1(t_n) + a_{n2}f_2(t_n) + \cdots + a_{nn}f_n(t_n),
		\end{split}\right.
	\end{equation*}
	hoặc dưới dạng ma trận nhân vector:
	\begin{equation*}
		\begin{pmatrix}
			f_1(t_{n+1})\\f_2(t_{n+1})\\\vdots\\f_n(t_{n+1})
		\end{pmatrix} = \begin{pmatrix}
			a_{11} & \cdots & a_{1n}\\a_{21} & \ddots & a_{2n}\\\vdots & \ddots & \vdots\\a_{n1} & \cdots & a_{nn}.
		\end{pmatrix}
	\end{equation*}
	Đặt $A = (a_{ij})_{i,j=1}^n$ \& ${\bf f}(t_{n+1}) = (f_1(t_{n+1}),f_2(t_{n+1}),\ldots,f_n(t_{n+1}))^\top$ thì
	\begin{equation*}
		{\bf f}(t_{n+1}) = A{\bf f}(t_n),\ \forall n\in\mathbb{N}.
	\end{equation*}
	Lặp:
	\begin{equation*}
		{\bf f}(t_m) = A{\bf f}(t_{m-1}) = A^2{\bf f}(t_{m-2}) = \cdots = A^m{\bf f}(t_0) = A^m\begin{pmatrix}
			\alpha_1\\\vdots\\\alpha_n
		\end{pmatrix},\ \forall m\in\mathbb{N}.
	\end{equation*}
	Đến đây ta cần tính $A^m$. Có 2 trường hợp:
	\begin{enumerate}
		\item {\bf Trường hợp $A$ chéo hóa được.} Tìm được ma trận $P\in M_n(\mathbb{R})$ khả nghịch thỏa $P^{-1}AP = \operatorname{diag}(\lambda_1(A),\ldots,\lambda_n(A))$ với $\{\lambda_i(A)\}_{i=1}^n$ là $n$ trị riêng (có thể gồm nghiệm riêng bội) của $A$, khi đó
		\begin{align*}
			A^m &= P\operatorname{diag}(\lambda_1^m(A),\ldots,\lambda_n^m(A))P^{-1},\ \forall m\in\mathbb{N},\\
			{\bf f}(t_m) &= P\operatorname{diag}(\lambda_1^m(A),\ldots,\lambda_n^m(A))P^{-1}\begin{pmatrix}
				\alpha_1\\\vdots\\\alpha_n
			\end{pmatrix},\ \forall m\in\mathbb{N}.
		\end{align*}
		\item {\bf Trường hợp $A$ không chéo hóa được.} Hy vọng $n$ đủ nhỏ hoặc có vài tính chất đơn giản để đơn giản được tính toán. Còn nếu không, trong trường hợp tổng quát, thì phải sử dụng binary exponentiation for square matrices bên Competitive Programming \& lập trình thì mới mong cứu nổi, see, e.g.
		
		NQBH. {\it Matrix Multiplication \& Fast Doubling Techniques in Competitive Programming}. \url{https://github.com/NQBH/advanced_STEM_beyond/blob/main/OLP_ICPC/matrix_multiplication/NQBH_matrix_multiplication.pdf}.
	\end{enumerate}	
\end{remark}

\begin{remark}[Bài toán chuyển trạng thái tổng quát]
	Sơ đồ chung của bài toán chuyển trạng thái hoặc thông tin tổng quát thường có dạng sau:
	\begin{equation*}
		{\bf s}_0\in F^{d_0}\xrightarrow{\phi_1}{\bf s}_1\in F^{d_1}\xrightarrow{\phi_2}{\bf s}_2\in F^{d_2}\xrightarrow{\phi_3}\cdots\xrightarrow{\phi_{n-1}}{\bf s}_{n-1}\in F^{d_{n-1}}\xrightarrow{\phi_n}{\bf s}_n\in F^{d_n}\xrightarrow{\phi_{n+1}}\cdots,
	\end{equation*}
	với $\{{\bf s}_n\}_{n=1}^\infty$ là dãy các trạng thái hoặc thông tin được mã hóa thành các vector, ${\bf s}_n$ là trạng thái hoặc vector thông tin ở thời điểm{\tt/}giai đoạn thứ $n$, \& $\{\phi_n\}_{n=1}^\infty$ là các ánh xạ, tuyến tính hoặc phi tuyến, e.g., ánh xạ tuyến tính, (artificial) neural networks (mạng thần kinh nhân tạo) (e.g., feedforward neural networks, recurrent neural networks, graph neural networks, etc., see, e.g., \cite{Jentzen_Kuckuck_Wurstemberger_DL}), etc., mang vai trò biến đổi trạng thái (state transform) hoặc vận chuyển thông tin (information transport) từ trạng thái đang xét (current state) sang trạng thái tiếp theo (next state). Bằng lặp hoặc quy nạp:
	\begin{equation*}
		{\bf s}_n = \phi_n({\bf s}_{n-1}) = \phi_n(\phi_{n-1}({\bf s}_{n-2})) = \cdots = (\phi_n\circ\phi_{n-1}\circ\cdots\phi_1)({\bf s}_0),\ \forall n\in\mathbb{N},
	\end{equation*}
	nên nếu muốn xác định vector trạng thái ${\bf s}_n$ thông qua vector trạng thái ban đầu ${\bf s}_0$, ta cần tính $\phi_n\circ\phi_{n-1}\circ\cdots\phi_1$.
\end{remark}

\subsection{Matrix equation -- Phương trình ma trận}
Đề Olympic Toán Sinh viên thường xét các dạng toán sau, với $0_n = (0)_{i,j=1}^n$ là ma trận không kích thước $n\times n$:
\begin{enumerate}
	\item {\it Có tồn tại hay không 1 ma trận $A\in M_n(\mathbb{R})$ thỏa phương trình ma trận $f(A) = 0_n$?}
	
	Nếu tồn tại thì chỉ cần chỉ ra 1 ví dụ cho nghiệm ma trận $A$ thỏa mãn $f(A) = 0_n$ là xong. Còn nếu không tồn tại thì phải chứng minh $f(A)\ne0_n$, $\forall A\in M_n(\mathbb{R})$, \& thường sử dụng phương pháp phản chứng, dùng các tính chất quen thuộc của ma trận như các tính chất của det, trace, $\ldots$.
	\item {\it Chứng minh không tồn tại ma trận $A\in M_n(\mathbb{R})$ thỏa phương trình ma trận $f(A) = 0_n$.}
	
	I.e., vế sau của bài toán 1.
	\item {\it Tìm tất cả các ma trận $A\in M_n(\mathbb{R})$ thỏa mãn phương trình ma trận $f(A) = 0_n$.}
	
	Bài toán này yêu cầu giải phương trình ma trận để tìm tất cả các nghiệm ma trận.
\end{enumerate}

\begin{baitoan}[\cite{VMS_VMC2025}, UIT-HCM]
	\item(a) Có tồn tại hay không ma trận $A\in M_3(\mathbb{R})$ thỏa ${\rm tr}\,A = 0$, $A^2 + A^\top = I_3$?
	\item(b) Tìm tất cả các ma trận thực $X\in M_2(\mathbb{R})$ thỏa
	\begin{equation*}
		X^3 - 3X^2 + \begin{pmatrix}
			2 & 2\\2 & 2
		\end{pmatrix} = 0_2.
	\end{equation*}
\end{baitoan}

\begin{remark}[Phân tích ý (b)]
	\item(i) Phương trình ma trận của (b) có chứa 1 ma trận $2\times2$ cụ thể:
	\begin{equation*}
		A = \begin{pmatrix}
			2 & 2\\2 & 2
		\end{pmatrix}.
	\end{equation*}
	Nhận thấy $\det A = 2\times2 - 2\times2 = 0$, nên ta có thể lấy $\det$ 2 vế để tận dụng tính chất $\det A = 0$. Hơn nữa,
	\begin{equation*}
		A - 4I_2 = \begin{pmatrix}
			2 & 2\\2 & 2
		\end{pmatrix} - 4\begin{pmatrix}
			1 & 0 \\0 & 1
		\end{pmatrix} = \begin{pmatrix}
			2 & -2\\-2 & 2
		\end{pmatrix}
	\end{equation*}
	cũng có $\det(A - 4I_2) = 2\times2 - (-2)\times(-2) = 0$ nên cũng có thể tận dụng tính chất này khi lấy $\det$ 2 vế, đương nhiên là sau khi thêm bớt số hạng $4I_n$ hợp lý.
	\item(ii) Nếu gặp các phương trình ma trận $f(X) = 0_n$ tương tự, với ẩn ma trận $X\in M_n(\mathbb{C})$, ta nên thử tìm các tính chất đặc biệt của các ma trận cụ thể trong phương trình ma trận đã cho, e.g., ma trận đó có định thức hoặc vết đặc biệt.
\end{remark}

\begin{tip}
	Để giải phương trình ma trận có dạng $f(A,A^\top) = 0_n$ với biến $A\in M_n(\mathbb{C})$, cố gắng biểu diễn $A^\top$ theo $A$, i.e., tìm hàm $g(x)$ để $A^\top = g(A)$, sau đó thay vào phương trình ma trận ban đầu để được $f(A,g(A)) = 0_n$, thường phương trình này sẽ là phương trình đa thức, cố gắng suy ra đa thức tối tiểu của $A$.
\end{tip}

\begin{baitoan}
	Cho $p\in\mathbb{C}[x]$. Tìm cách giải phương trình ma trận $p(A) = 0_n$ với biến ma trận $A\in M_n(\mathbb{C})$.
\end{baitoan}

\begin{baitoan}
	1 tổng quát hóa của ý (b) bài toán trên có dạng: Cho $p\in\mathbb{C}[x]$. Giải phương trình ma trận
	\begin{equation*}
		P(A,B)\coloneqq\sum_{i=1}^n a_iA^i + B = 0_n
	\end{equation*}
	với biến ma trận $A\in M_n(\mathbb{R})$, còn ma trận $B\in M_n(\mathbb{R})$ được cho trước (với vai trò như hệ số tự do trong 1 đa thức 1 biến $p\in\mathbb{R}[x]$).
\end{baitoan}

\subsection{Sequences -- Dãy số}
Cho 1 dãy số $\{a_n\}_{n=1}^\infty$. Đề Olympic Toán Sinh viên thường xét các dạng toán sau:
\begin{enumerate}
	\item Dãy số $\{a_n\}_{n=1}^\infty$ có hội tụ hay không?
	\item Chứng minh dãy số $\{a_n\}_{n=1}^\infty$ hội tụ hoặc phân kỳ.
	\item Tính giới hạn $\lim_{n\to\infty} a_n$.
\end{enumerate}

\begin{dinhly}
	Nếu 1 dãy tăng ngặt (hoặc không giảm) \& bị chặn trên thì hội tụ. Nếu 1 dãy giảm ngặt (hoặc không tăng) thì hội tụ.
\end{dinhly}

\begin{tip}
	Nếu dãy số có số hạng $u_n$ có công thức tổng quát là 1 tích, thì có thể dùng hàm $ln$ để biến tích thành tổng bằng cách đặt $v_n\coloneqq\ln u_n$.
\end{tip}

\begin{tip}[Kỹ thuật làm trội{\tt/}giảm]
	Nếu công thức có xuất hiện các số vô tỷ, hoặc không chính phương, có thể thay bằng số chính phương để tiện phân tích nhân tử. E.g., B1 bảng A Giải tích 2025, có thể làm trội như sau:
	\begin{equation*}
		a_n = \prod_{i=1}^n \left(1 - \frac{\pi}{(i + 1)^2}\right) > \prod_{i=1}^n \left(1 - \frac{4}{(i + 1)^2}\right) = \prod_{i=1}^n \left(1 - \frac{1}{i + 1}\right)\left(1 + \frac{1}{i + 1}\right)
	\end{equation*}
	vì $\pi$ là số vô tỷ, khá khó chịu, nên làm trội bởi $\pi < 4$ với $4$ là 1 số chính phương gần nhất.
\end{tip}

\begin{baitoan}[VMC2025GTA1B1]
	Cho $\{a_n\}_{n=1}^\infty$ là 1 dãy số thực được xác định bởi
	\begin{equation*}
		a_n = \prod_{i=1}^n \left(1 - \frac{\pi}{(i + 1)^2}\right) = \left(1 - \frac{\pi}{2^2}\right)\cdots\left(1 - \frac{\pi}{(n + 1)^2}\right),\ \forall n\in\mathbb{N}^\star.
	\end{equation*}
	\item(a) Chứng minh $\{a_n\}_{n=1}^\infty$ hội tụ.
	\item(b) Giới hạn của dãy số $\{a_n\}_{n=1}^\infty$ có bằng $0$ được không? Vì sao?
\end{baitoan}

\begin{tip}
	Nếu cần xét dấu đạo hàm $f'(x)$, có thể phải sử dụng bất đẳng thức Cauchy--Schwartz:
	\begin{equation*}
		\left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right)\ge\left(\sum_{i=1}^n a_ib_i\right)^2,\ \forall a_i,b_i\in\mathbb{R},\ \forall i\in[n].
	\end{equation*}
	Bất đẳng thức Cauchy--Schwartz là hệ quả trực tiếp của hằng đẳng thức Lagrange:
	\begin{equation*}
		\left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right) = \left(\sum_{i=1}^n a_ib_i\right)^2 + \sum_{1\le i < j\le n} (a_ib_j - a_jb_i)^2\ge\left(\sum_{i=1}^n a_ib_i\right)^2.
	\end{equation*}
	Nhưng cần phải nhớ kỹ công thức hằng đẳng thức Lagrange, không thì cứ sử dụng bất đẳng thức Cauchy--Schwartz.
\end{tip}

\begin{tip}
	Nếu dính bài cần đánh giá $\cos$ thì sử dụng bất đẳng thức
	\begin{equation*}
		\cos x\ge1 - \frac{x^2}{2},\ \forall x\ge0.
	\end{equation*}
\end{tip}

\begin{dinhly}[Darboux]
	Nếu đạo hàm $f'$ của 1 hàm số $f$ nhận 2 giá trị $a,b\in\mathbb{R}$ thì $f'$ cũng nhận tất cả các giá trị thuộc $[a,b]$.
\end{dinhly}
I.e.,
\begin{equation*}
	a,b\in{\rm Im}(f')\Leftrightarrow[a,b]\subset{\rm Im}(f').
\end{equation*}

\begin{tip}[Kỹ thuật liên tục hóa bài toán rời rạc]
	Đối với các bài yêu cầu tính giới hạn của dãy số $\{a_n\}_{n=1}^\infty$, nên chuyển từ phiên bản ``rời rạc'' thành phiên bản ``liên tục'' bằng cách tìm hàm số $f\in C(\mathbb{R})$ để $a_{n+1} = f(a_n)$, $\forall n\in\mathbb{N}^\star$ hoặc từ 1 số hạng nào đó trở đi, ngoại trừ vài số hạng đầu, để khảo sát tính chất của hàm số $f$, từ đó suy ra tính chất của dãy số $\{a_n\}_{n=1}^\infty$. Lợi thế của kỹ thuật liên tục hóa là ở phiên bản liên tục, có các công cụ giải tích mạnh như đạo hàm, tích phân, khảo sát hàm số, etc.
\end{tip}

\begin{baitoan}[ĐHGTVT, \cite{VMS_VMC2025}, 7., p. 38]
	Cho dãy số $\{x_n\}_{n=1}^\infty$ thỏa
	\begin{equation*}
		x_1 = 2025,\ x_{n+1} = \sqrt{5} + \frac{2x_n}{\sqrt{x_n^2 - 4}},\ \forall n\in\mathbb{N}^\star.
	\end{equation*}
	Chứng minh $\{x_n\}_{n=1}^\infty$ hội tụ \& tính giới hạn của dãy số đó.
\end{baitoan}

\begin{proof}[Giải]
	Có $x_1 = 2025 > 2 + \sqrt{5}$ \&
	\begin{equation*}
		x_{n+1} = \sqrt{5} + \frac{2x_n}{\sqrt{x_n^2 - 4}} > \sqrt{5} + \frac{2x_n}{x_n} = 2 + \sqrt{5},\ \forall n\in\mathbb{N}^\star.
	\end{equation*}
	Xét hàm số $f(x) = \sqrt{5} + \frac{2x}{\sqrt{x^2 - 4}}$, $\forall x\in(2 + \sqrt{5},\infty)$, có
	\begin{equation*}
		f'(x) = -\frac{8}{(x^2 - 4)\sqrt{x^2 - 4}} < 0,\ \forall x\in(2 + \sqrt{5},\infty),
	\end{equation*}
	thỏa
	\begin{equation*}
		|f'(x)| = \frac{8}{(x^2 - 4)\sqrt{x^2 - 4}} < a\coloneqq\frac{8}{(5 + 4\sqrt{5})\sqrt{5 + 4\sqrt{5}}}\approx0.1536371431,\ \forall x\in(2 + \sqrt{5},\infty).
	\end{equation*}
	Tới đây có 2 cách để xử lý:
	\begin{enumerate}
		\item Với mỗi $n\in\mathbb{N}^\star$, vì $x_{n+1} = f(x_n)$, $f(2\sqrt{5}) = 2\sqrt{5}$, sử dụng định lý Lagrange, tồn tại $c_n$ nằm giữa $x_{n+1}$ \& $2\sqrt{5}$ thỏa $f(x_n) - f(2\sqrt{5}) = f'(c_n)(x_n - 2\sqrt{5})$, nên
		\begin{align*}
			|x_{n+1} - 2\sqrt{5}| &= |f(x_n) - f(2\sqrt{5})| = |f'(c_n)(x_n - 2\sqrt{5})| = |f'(c_n)||x_n - 2\sqrt{5})|\le a|x_n - 2\sqrt{5}|\\
			&\le a^2|x_{n-1} - 2\sqrt{5}|\le\cdots\le a^n|x_1 - 2\sqrt{5}| = a^n|2025 - 2\sqrt{5}|\to0\mbox{ as } n\to\infty,
		\end{align*}
		suy ra $\lim_{n\to\infty} x_n = 2\sqrt{5}$.
		\item Sử dụng định lý Lagrange, suy ra dãy $\{x_n\}_{n=1}^\infty$ là dãy Cauchy, nên $\{x_n\}_{n=1}^\infty$ hội tụ về giá trị $l\ge2 + \sqrt{5}$. Cho $n\to\infty$ trong phương trình truy hồi của dãy $\{x_n\}_{n=1}^\infty$, được phương trình $l = \sqrt{5} + \frac{2l}{\sqrt{l^2 - 4}}$ có nghiệm duy nhất $l = 2\sqrt{5}$ nên $\lim_{n\to\infty} x_n = 2\sqrt{5}$.
	\end{enumerate}
\end{proof}

\begin{remark}
	Lập luận chính của chứng minh trên dựa vào nguyên lý ánh xạ co (contraction mapping theorem).
\end{remark}

\begin{dinhly}
	Cho $f\in D([a,b])$ có $|f'(x)|\le c$, $\forall x\in(a,b)$. Dãy số $\{x_n\}_{n=1}^\infty$ xác định bởi
	\begin{equation*}
		x_1 = \alpha\in[a,b],\ x_{n+1} = f(x_n),\ \forall n\in\mathbb{N}^\star,
	\end{equation*}
	hội tụ \& có giới hạn là nghiệm duy nhất của phương trình $f(x) = x$, i.e., điểm bất động của $f$.
\end{dinhly}

\subsection{Continuous function -- Hàm số liên tục}
{\it Intuition.} Hàm số liên tục là hàm mà khi ta vẽ đồ thị của hàm đó, thì ta không được phép nhấc cây viết lên. Nếu nhấc lên thì hàm số sẽ bị gián đoạn (discontinuities) tại điểm nhấc lên đó, nên hàm số đó không liên tục (discontinuous function).

Gọi $F_{\rm elm}(F)$ là tập hợp các {\it hàm số sơ cấp (elementary function)} trên trường $F$, see, e.g., \href{https://en.wikipedia.org/wiki/Elementary_function}{Wikipedia{\tt/}elementary functions}.

\end{dinhnghia}

\begin{bode}[Định lý kẹp cho giới hạn hàm số]
	Cho $f,g:D\to\mathbb{R}$ là 2 hàm số thỏa mãn $|f(x)|\le g(x)$, $\forall x\in D$, $x_0\in D$, \& $\lim_{x\to x_0} g(x) = 0$. Khi đó $\lim_{x\to x_0} f(x) = 0$.
\end{bode}

\begin{tip}
	Để chứng minh hàm số $f(x)$ không liên tục tại $x = x_0$, có thể tìm 2 dãy số $\{x_n\}_{n=1}^\infty,\{y_n\}_{n=1}^\infty$ cùng hội tụ về $x_0$ nhưng giới hạn của 2 dãy $\{f(x_n)\}_{n=1}^\infty,\{f(y_n)\}_{n=1}^\infty$lại khác nhau, i.e.,
	\begin{equation*}
		\left\{\begin{split}
			\lim_{n\to\infty} x_n &= \lim_{n\to\infty} y_n = x_0,\\
			\lim_{n\to\infty} f(x_n)&\ne\lim_{n\to\infty} f(y_n).
		\end{split}\right.		
	\end{equation*}
	Khi đó, theo định nghĩa liên tục điểm của hàm số, $f$ không liên tục (gián đoạn) tại $x = x_0$.
\end{tip}

\subsection{Derivatives of differentiable functions -- Các đạo hàm của các hàm số khả vi}

\paragraph{Notations.} Với $\Omega\subset\mathbb{R}^d$ là 1 tập mở, Ký hiệu ${\cal D}(\Omega)$ là tập tất cả các hàm số $f\in C(\overline{\Omega})$, i.e., $f$ liên tục trên $\overline{\Omega}$ \& khả vi trên $\Omega$, i.e., đạo hàm $f'(x)$ tồn tại \& xác định tại mọi điểm $x\in\Omega$ nhưng chưa chắc $f'\in C(\Omega)$, i.e., $f'(x)$ có thể liên tục trên $\Omega$ hoặc không liên tục.

\begin{dinhnghia}[Đạo hàm]
	
\end{dinhnghia}

\begin{example}
	A common example of a function that is differentiable but not $C^1$ is
	\begin{equation*}
		f(x) = \left\{\begin{split}
			&x^2\sin\frac{1}{x}&&\mbox{if }x\ne0,\\
			&0&&\mbox{if }x = 0,
		\end{split}\right.
	\end{equation*}
	where the derivative exists everywhere but is not continuous at $x = 0$.
\end{example}
Bài toán sau hoàn toàn tương tự ví dụ vừa nêu:

\begin{baitoan}[ĐHGTVT, \cite{VMS_VMC2025}, 11., p. 39]
	Cho hàm số
	\begin{equation*}
		f(x) = \left\{\begin{split}
			&x^2\sin\frac{1}{x^2}&&\mbox{if }x\ne0,\\
			&a&&\mbox{if }x = 0,
		\end{split}\right.
	\end{equation*}
	\item(a) Tìm $a\in\mathbb{R}$ để $f$ liên tục tại $x = 0$.
	\item(b) Với $a$ vừa tìm được, tính $f'(x)$.
	\item(c) Xét tính liên tục của hàm số $f'(x)$.
\end{baitoan}

\begin{baitoan}[ĐHGTVT, \cite{VMS_VMC2025}, 12., p. 39]
	Giả sử hàm số $f:\mathbb{R}\to\mathbb{R}$ thỏa
	\begin{equation*}
		f(x + y) = f(x) + f(y),\ \forall x,y\in\mathbb{R}.
	\end{equation*}
	Chứng minh:
	\item(a) $f(nx) = nf(x)$, $\forall n\in\mathbb{Z}$.
	\item(b) $f$ liên tục tại 1 điểm $\Leftrightarrow$ $f$ liên tục trên $\mathbb{R}$.
	\item(c) $f$ liên tục $\Leftrightarrow f(x) = ax$ với $a\in\mathbb{R}$.
\end{baitoan}

\begin{proof}[Giải]
	(a) Gọi $S(a,b)$ là phép thay (substitution) $x = a,y = b$ vào phương trình hàm đang xét. $S(0,0)\Rightarrow f(0) = f(0 + 0) = f(0) + f(0)\Rightarrow f(0) = 0$. $S(x,x)\Rightarrow f(2x) = f(x + x) = f(x) + f(x) = 2f(x)$. $S(2x,x)\Rightarrow f(3x) = f(2x + x) = f(2x) + f(x) = 2f(x) + f(x) = 3f(x)$, quy nạp được $S((n - 1)x,x)\Rightarrow  f(nx) = f((n - 1)x + x) = f((n - 1)x) + f(x) = (n - 1)f(x) + f(x) = nf(x)$, $\forall n\in\mathbb{N}$, $\forall x\in\mathbb{R}$. Hơn nữa, $S(x,-x)\Rightarrow f(0) = f(x + (-x)) = f(x) + f(-x)\Rightarrow f(-x) = -f(x)$, $\forall x\in\mathbb{R}$. Nên $f(nx) = nf(x)$, $\forall n\in\mathbb{Z}$, $\forall x\in\mathbb{R}$.
	\item(b) Chiều đảo: $f$ liên tục trên $\mathbb{R}\Rightarrow f$ liên tục tại 1 điểm là hiển hiển (tổng quát $\Rightarrow$ cụ thể). Ta cần chứng minh chiều thuận: $f$ liên tục tại 1 điểm $\Rightarrow f$ liên tục trên $\mathbb{R}$. Giả sử $f$ liên tục tại 1 điểm $c\in\mathbb{R}$ cố định, suy ra $\lim_{x\to c} f(x) = f(c)$, i.e., $\forall\varepsilon\in(0,\infty)$, $\exists\delta(\varepsilon)\in(0,\infty)$ thỏa $|x - c| < \delta(\varepsilon)\Rightarrow|f(x) - f(c)| < \varepsilon$. Khi đó, $\forall x,x_0\in\mathbb{R}$ thỏa $|x - x_0| < \delta(\varepsilon)$ thì $|(x - x_0 + c) - c| < \delta(\varepsilon)$, suy ra $|f(x) - f(x_0)| = |f(x - x_0)| = |f(x - x_0) + f(c) - f(c)| = |f(X - x_0 + c) - f(c)| < \varepsilon$, nên $\lim_{x\to x_0} f(x) = f(x_0)$, i.e., $f$ liên tục tại $x_0\in\mathbb{R}$ bất kỳ, i.e., $f$ liên tục trên $\mathbb{R}$.
	\item(c) Chứng minh mở rộng từ tập $\mathbb{Z}$ lên $\mathbb{Q}$, rồi nâng lên $\mathbb{R}$ nhờ việc $\mathbb{Q}$ trù mật trong $\mathbb{R}$, see, e.g., \cite{Chung_day_so, Chung_pth_da_thuc}.
\end{proof}

\begin{remark}[Phương trình hàm Cauchy]
	Bài toán yêu cầu tìm tất cả các hàm cộng tính, i.e., $f(x + y) = f(x) + f(y)$ là phương trình hàm Cauchy quen thuộc với học sinh chuyên Toán.
\end{remark}

\begin{tip}
	Đối với các hàm số không phải là hàm sơ cấp, đặc biệt là hàm ghép lại của các hàm sơ cấp, i.e.,
	\begin{equation*}
		f(x) = \sum_{i=1}^{n-1} f_i(x){\bf1}_{(a_i,a_{i+1}]}(x) + f_n(x){\bf1}_{(a_n,a_{n+1})}(x) = \left\{\begin{split}
			&f_1(x)&&\mbox{if } x\in(a_1,a_2],\\
			&f_2(x)&&\mbox{if } x\in(a_2,a_3],\\
			&\ldots\\
			&f_{n-1}(x)&&\mbox{if } x\in(a_{n-1},a_n],\\
			&f_n(x)&&\mbox{if } x\in(a_n,a_{n+1}),
		\end{split}\right.
	\end{equation*}
	với $f_i\in F_{\rm elm}(\mathbb{R})$, $\forall i\in[n]$, $a_1,a_{n+1}\in\overline{\mathbb{R}}$, $a_i\in\mathbb{R},\ \forall i\in[[2,n]]$ thỏa $a_i < a_{i+1}$, $\forall i\in[n]$, thì nên dùng định nghĩa của đạo hàm thông qua giới hạn để tính $f'(x)$.
\end{tip}

\begin{tip}[Đặt số thành biến]

\begin{baitoan}[VMC2025GTB2]
	Cho $f:(0,\infty)\to\mathbb{R}$ là 1 hàm số được xác định bởi
	\begin{equation*}

\subsection{Mean-valued theorems -- Các định lý giá trị trung bình}

\begin{dinhly}[Fermat]
	\label{dl: Fermat}
	Cho $f(x)$ xác định trên $(a,b)$. Nếu $f(x)$ đạt cực trị tại $x_0$ \& khả vi tại $x_0$ thì $f'(x_0) = 0$.
\end{dinhly}

\begin{remark}
	Chiều ngược lại không đúng vì $f$ có thể đạt cực trị tại các điểm $x\in D_f$ mà $f$ không khả vi tại các điểm đó. Gọi tập nghiệm của $f'$ là
	\begin{equation*}
		Z(f')\coloneqq(f')^{-1}(\{0\}) = \{x\in(a,b);f'(x) = 0\},
	\end{equation*}
	\& tập các điểm cực trị (extremum or extreme point) của $f$ là
	\begin{equation*}
		{\rm expt}(f) = \{x\in D_f;x\mbox{ is an extreme point of } f\},
	\end{equation*}
	thì định lý Fermat \ref{dl: Fermat} chỉ khẳng định rằng nếu $f(x)$ đạt cực trị tại $x_0$ \& khả vi tại $x_0$ thì $f'(x_0) = 0$, i.e., $x_0\in Z(f')$.
\end{remark}

\begin{dinhly}[Rolle]
	\label{dl: Rolle}
	Giả sử $f(x)$ xác định \& liên tục trên $[a,b]$ hữu hạn, khả vi trên $(a,b)$ \& $f(a) = f(b)$. Khi đó tồn tại $c\in(a,b)$ sao cho $f'(c) = 0$.
\end{dinhly}
{\sf Intuition.} 1 sợi dây cố định 2 đầu mút tại chiều cao $h = f(a) = f(b)$. Trong cả 2 trường hợp sợi dây hướng lên hoặc hướng xuống từ mốc trái, thì sợi đây đó phải đi theo cách ngược lại theo chiều cao để trở lại chiều cao $h$ ban đầu, thì chỗ sợi dây ``đổi chiều lên{\tt/}xuống $x = c\in(a,b)$ chính là chỗ mà $f'$ triệt tiêu, i.e., $f'(c) = 0$. Sợi dây không đứt này tượng trưng cho hàm liên tục.

\begin{dinhly}[Lagrange]
	\label{dl: Lagrange}
	Cho $f(x)$ xác định \& liên tục trên $[a,b]$, khả vi trên $(a,b)$. Khi đó tồn tại $c\in(a,b)$ sao cho
	\begin{equation}
		\label{Lagrange}
		f'(c) = \dfrac{f(b) - f(a)}{b - a}.
	\end{equation}
\end{dinhly}

\begin{remark}
	Định lý \ref{dl: Lagrange} chỉ đảm bảo sự tồn tại (existence) của ít nhất 1 số $c\in(a,b)$ thỏa mãn \eqref{Lagrange}, không đảm bảo tính duy nhất (uniqueness), nên không thể định nghĩa $c$ như 1 hàm $g(f,a,b)$ \& thỏa mãn \eqref{Lagrange}, mà chỉ có thể định nghĩa tập hợp
	\begin{equation*}
		C_{\rm Lagrange}(f,x,y)\coloneqq\left\{c\in(x,y);f'(c) = \dfrac{f(x) - f(y)}{x - y}\right\},\ \forall f\in D([a,b]),\ \forall x,y\in[a,b], x\ne y.
	\end{equation*}
	Định lý Lagrange \ref{dl: Lagrange} chỉ đảm bảo rằng $C_{\rm Lagrange}(f,x,y)\ne\emptyset$, không hề nói tập $C_{\rm Lagrange}(f,x,y)$ chỉ có 1 phần tử duy nhất, e.g., dễ thấy điều này sai ngay đối với các hàm hằng.
\end{remark}

\begin{dinhly}[Cauchy]
	Cho $f(x),g(x)$ liên tục trên $[a,b]$, khả vi trên $(a,b)$, $g'(x)\ne0$, $\forall x\in(a,b)$. Khi đó tồn tại $c\in(a,b)$ sao cho
	\begin{equation*}
		\dfrac{f(b) - f(a)}{g(b) - g(a)} = \dfrac{f'(c)}{g'(c)}.
	\end{equation*}
\end{dinhly}

\begin{remark}
	Nếu ``chứng minh'' 1 cách quá ngây thơ bằng cách thêm bớt nhân tử $b - a$ như sau:
	\begin{equation*}
		\dfrac{f(b) - f(a)}{g(b) - g(a)} = \dfrac{\dfrac{f(b) - f(a)}{b - a}}{\dfrac{g(b) - g(a)}{b - a}},
	\end{equation*}
	rồi áp dụng định lý Lagrange 2 lần thì chỉ đảm bảo tồn tại $c,d\in(a,b)$ thỏa
	\begin{equation*}
		f'(c) = \frac{f(b) - f(a)}{b - a},\ g'(d) = \frac{g(b) - g(a)}{b - a},
	\end{equation*}
	nhưng không đảm bảo $c = d$ để có 1 ``tiếng nói chung'' kiểu $\frac{f'(c)}{g'(d)}$.	
\end{remark}

\begin{dinhly}[Darboux]
	\label{dl: Darboux}
	Nếu hàm số $f(x)$ khả vi trên $(a,b)$, $\alpha,\beta\in(a,b)$ thì $f'(x)$ nhận mọi giá trị trung gian giữa $f'(\alpha)$ \& $f'(\beta)$.
\end{dinhly}
I.e., định lý Darboux có thể được viết như sau:
\begin{equation*}
	(\min\{f'(\alpha),f'(\beta)\},\max\{f'(\alpha),f'(\beta)\})\subset f'((a,b)),\ \forall f(x)\mbox{ differentiable on }(a,b),\ \forall\alpha,\beta\in(a,b).
\end{equation*}

\begin{baitoan}[\cite{Quoc_Long_Dat_Nam_VMC}, 3.14., p. 152]
	Cho $f:\left[-\frac{\pi}{2},\frac{\pi}{2}\right]\to[-1,1]$ là 1 hàm khả vi có đạo hàm liên tục \& không âm ($f'\ge0$). Chứng minh tồn tại $x_0\in\left(-\frac{\pi}{2},\frac{\pi}{2}\right)$ sao cho $(f(x_0))^2 + (f'(x_0))^2\le1$.
\end{baitoan}

\begin{tip}[Giải các bài toán về các định lý giá trị trung bình]
	Đối với các bài toán về các định lý giá trị trung bình, chọn đúng hàm cần xét, thường có thể đoán dựa vào (các) giả thiết của bài toán, để áp dụng đúng định lý giá trị trung bình là mấu chốt để giải. Nên các bài toán áp dụng (các) giá trị định lý trung bình (vài lần) dạng này khá thiên về kiểu đánh đố.
\end{tip}

\begin{baitoan}[UIT, 15., \cite{VMS_VMC2025}, p. 39]
	Cho hàm $f\in C([0,1])$ thỏa $\int_0^1 f(x)\,{\rm d}x = \int_0^1 xf(x)\,{\rm d}x$. Chứng minh tồn tại $c\in(0,1)$ thỏa
	\begin{equation*}
		(1 - c)f(c) + c\int_0^c f(x)\,{\rm d}x = 0.
	\end{equation*}
\end{baitoan}

\begin{proof}
	Đặt
	\begin{equation*}
		g(x)\coloneqq x\int_0^x f(t)\,{\rm d}t - \int_0^x tf(t)\,{\rm d}t,\ \forall x\in[0,1],
	\end{equation*}
	thì
	\begin{equation*}
		g'(x) = \int_0^x f(t)\,{\rm d}t + xf(x) - xf(x) = \int_0^x f(t)\,{\rm d}t,\ \forall x\in(0,1).
	\end{equation*}
	Từ giả thiết, có $g(0) = 0$, $g(1) = \int_0^1 f(t)\,{\rm d}t  - \int_0^1 tf(t)\,{\rm d}t = 0$, theo định lý Rolle \ref{dl: Rolle}, tồn tại $\alpha\in(0,1)$ thỏa $g'(\alpha) = 0$, i.e., $\int_0^\alpha f(t)\,{\rm d}t = 0$. Xét hàm
	\begin{equation*}
		h(x)\coloneqq\frac{e^{-x}}{1 - x}\int_0^x f(t)\,{\rm d}t,\ \forall x\in[0,\alpha],
	\end{equation*}
	thì $h(0) = h(\alpha) = 0$ \&
	\begin{equation*}
		h'(x) = \frac{xe^{-x}}{(1 - x)^2}\int_0^x f(t)\,{\rm d}t + \frac{e^{-x}}{1 - x}f(x) = \frac{e^{-x}}{(1 - x)^2}\left[x\int_0^x f(t)\,{\rm d}t + (1 - x)f(x)\right],\ \forall x\in(0,1),
	\end{equation*}
	nên theo định lý Rolle \ref{dl: Rolle}, tồn tại $c\in(0,\alpha)\subset(0,1)$ thỏa $h'(c) = 0$, i.e., đpcm.
\end{proof}

\begin{nhanxet}
	Lời giải trên áp dụng 2 lần định lý Rolle lần lượt cho 2 hàm $g(x),h(x)$. Công thức của hàm $g(x)$ khá tự nhiên nhờ vào giả thiết $\int_0^1 f(x)\,{\rm d}x = \int_0^1 xf(x)\,{\rm d}x$, nhưng công thức của hàm $h$ thì phức tạp hơn, \& phải dựa vào phương trình cần chứng minh để dự đoán nhân tử $\frac{e^{-x}}{1 - x}$ của $h(x)$.
\end{nhanxet}

\begin{baitoan}
	Mở rộng phạm vi áp dụng của các lý luận \& kỹ thuật trong lời giải trên.
\end{baitoan}

\begin{proof}
	Xét 2 hàm $f,g\in C([a,b])$ với $a,b\in\mathbb{R}$, $a < b$, thỏa
	\begin{equation*}
		g(b)\int_a^b f(x)\,{\rm d}x = \int_a^b f(x)g(x)\,{\rm d}x.
	\end{equation*}
	Xét hàm:
	\begin{align*}
		h(x)\coloneqq g(x)\int_a^x f(t)\,{\rm d}t - \int_a^x f(t)g(t)\,{\rm d}t,\ \forall x\in[0,1],
	\end{align*}
	có $h(a) = 0$ (nhờ tích phân $\int_a^a F(x)\,{\rm d}x = 0$ với mọi hàm $F\in R([a,b])$) \& $h(b) = g(b)\int_a^b f(t)\,{\rm d}t - \int_a^b f(t)g(t)\,{\rm d}t = 0$ nhờ vào giả thiết, \& có đạo hàm
	\begin{equation*}
		h'(x) = g'(x)\int_a^x f(t)\,{\rm d}t + g(x)f(x) - f(x)g(x) = g'(x)\int_a^x f(t)\,{\rm d}t,\ \forall x\in(a,b),
	\end{equation*}
	theo định lý Rolle \ref{dl: Rolle}, tồn tại $c\in(a,b)$ thỏa $h'(c) = 0$, i.e., $g'(c)\int_a^c f(t)\,{\rm d}t = 0$. Xét 2 trường hợp:
	\begin{itemize}
		\item {\it Trường hợp $g,a,b$ thỏa $g'(x)\ne0$, $\forall x\in(a,b)$:} Có $g'(c)\ne0$, nên $h'(c) = 0\Leftrightarrow\int_a^c f(t)\,{\rm d}t = 0$. Xét tiếp hàm
		\begin{equation*}
			l(x)\coloneqq k(x)\int_a^x f(t)\,{\rm d}t,\ \forall x\in[a,c],
		\end{equation*}
		có $l(a) = l(c) = 0$, \&
		\begin{equation*}
			l'(x) = k'(x)\int_a^x f(t)\,{\rm d}t + k(x)f(x),\ \forall x\in(a,c),
		\end{equation*}
		nên theo định lý Rolle \ref{dl: Rolle}, tồn tại $d\in(a,c)$ thỏa $l'(d) = 0$, i.e.,
		\begin{equation*}
			k'(d)\int_a^d f(t)\,{\rm d}t + k(d)f(d) = 0.
		\end{equation*}
		Tùy theo phương trình cần phải chứng minh
		\begin{equation*}
			A_{\rm free}(d)f(d) + A_{\rm int}(d)\int_a^d f(x)\,{\rm d}x = 0,
		\end{equation*}
		cụ thể là 2 hệ số mà ta sẽ chọn được hàm số $k(x)$ phù hợp, bằng cách giải hoặc chỉ cần tìm 1 nghiệm thỏa mãn phương trình vi phân bậc nhất
		\begin{equation*}
			\frac{k'(x)}{A_{\rm int}(x)} = \frac{k(x)}{A_{\rm free}(x)},
		\end{equation*}
		hay
		\begin{equation*}
			A_{\rm free}(x)k'(x) = A_{\rm int}(x)k(x),\ \forall x\in(a,b).
		\end{equation*}
		\item {\it Trường hợp $g,a,b$ chưa chắc thỏa $g'(x)\ne0$, $\forall x\in(a,b)$:} Ta phải xét hàm
		\begin{equation*}
			l(x)\coloneqq k(x)g'(x)\int_a^x f(t)\,{\rm d}t,\ \forall x\in[a,c],
		\end{equation*}
		có $l(a) = l(c) = 0$, \&
		\begin{equation*}
			l'(x) = (k'(x)g'(x) + k(x)g''(x))\int_a^x f(t)\,{\rm d}t + k(x)g'(x)f(x),\ \forall x\in(a,c),
		\end{equation*}
		nên theo định lý Rolle \ref{dl: Rolle}, tồn tại $d\in(a,c)$ thỏa $l'(d) = 0$, i.e.,
		\begin{equation*}
			(k'(d)g'(d) + k(d)g''(d))\int_a^d f(t)\,{\rm d}t + k(d)g'(d)f(d) = 0.
		\end{equation*}
		Tùy theo phương trình cần phải chứng minh
		\begin{equation*}
			A_{\rm free}(d)f(d) + A_{\rm int}(d)\int_a^d f(x)\,{\rm d}x = 0,
		\end{equation*}
		cụ thể là 2 hệ số mà ta sẽ chọn được hàm số $k(x)$ phù hợp, bằng cách giải hoặc chỉ cần tìm 1 nghiệm thỏa mãn phương trình vi phân bậc nhất
		\begin{equation*}
			\frac{k'(x)g'(x) + k(x)g''(x)}{A_{\rm int}(x)} = \frac{k(x)g'(x)}{A_{\rm free}(x)},\ \forall x\in(a,b),
		\end{equation*}
		hay
		\begin{equation*}
			A_{\rm free}(x)g'(x)k'(x) = \left(A_{\rm int}(x)g'(x) - A_{\rm free}(x)g''(x)\right)k(x),\ \forall x\in(a,b).
		\end{equation*}
	\end{itemize}
	Ta cũng có thể mở rộng giả thiết \& kết luận cho bài toán gốc bằng cách xét hàm
	\begin{align*}
		h_1(x)\coloneqq g(x)\int_a^x f(t)\,{\rm d}t + \int_a^x f(t)g(t)\,{\rm d}t,\ \forall x\in[0,1],
	\end{align*}
	có đạo hàm
	\begin{equation*}
		h_1'(x) = 2f(x)g(x) + g'(x)\int_a^x f(t)\,{\rm d}t,\ \forall x\in(a,b),
	\end{equation*}
	nên tính toán sẽ phức tạp hơn, đặc biệt là phương trình vi phân ở bước cuối, nhưng lý luận chính dựa vào việc áp dụng 2 lần định lý Rolle vẫn tương tự như quá trình xử lý đối với hàm $h(x)$ ở trên.
\end{proof}

\begin{baitoan}[UIT, 16., \cite{VMS_VMC2025}, p. 39]
	Với mỗi $x\in[0,\infty)$, gọi $z = z(x)$ là nghiệm dương của phương trình $z^4 + xz^2 = 2025$. Tính $\int_0^{2024} z^3\,{\rm d}x$.
\end{baitoan}

\section{Combinatorics -- Tổ Hợp}
Các bài thi Tổ hợp thì nói chung hên xui, nên bắt đầu thử các trường hợp nhỏ của $n$ nếu đề yêu cầu giải quyết cho trường hợp $n$ tổng quát, để tạo ``cảm giác'' đối với cấu trúc của bài toán Tổ hợp.

\begin{tip}[Extremal combinatorics -- Tổ hợp cực hạn]
	Nếu bài toán Tổ hợp yêu cầu tìm các giá trị lớn nhất hoặc nhỏ nhất, thì thường đó sẽ là bài toán Tổ hợp Cực biên, nên thử các cấu hình ở biên để có thêm thông tin.
\end{tip}

\begin{baitoan}[VMC2025ĐSA3B4]
	Điền 9 số $1,2,\ldots,9$ vào 9 ô đơn vị của 1 bảng $3\times3$, mỗi ô 1 số khác nhau. Xét 6 số có 3 chữ số được tạo thành từ các hàng (đọc từ trái qua phải) \& các cột (đọc từ trên xuống dưới).
	\item(a) Có thể điền sao cho $6$ số nhận được đều chia hết cho $9$ không?
	\item(b) Gọi $M$ là số lớn nhất trong $6$ số nhận được bằng 1 cách điền bất kỳ. Tìm $\min M$.
\end{baitoan}

\begin{tip}
	Bắt đầu từ các phần tử cực biên, nhỏ nhất \& lớn nhất để thu hẹp các trường hợp có thể xảy ra.
\end{tip}

\begin{proof}[Giải]
	(a) Tìm 1 cấu hình thỏa mãn là xong, e.g.,
	\begin{equation*}
		\begin{bmatrix}
			3 & 5 & 1\\4 & 6 & 8\\2 & 7 & 9
		\end{bmatrix}.
	\end{equation*}
	Cấu hình này có thể tìm bằng cách bắt đầu điền số $9$ vào 1 ô trong 4 góc.
\end{proof}

\begin{baitoan}
	Mở rộng bài VMC2025ĐSA3B4 khi điền $[n^2]$ vào $n^2$ ô đơn vị của 1 bảng $n\times n$.
\end{baitoan}
