#include <iostream>
using namespace std;

int main() {
    float a, b, c;
    cin >> a >> b >> c;

    float max = (a > b ? a : b);
    max = (max > c ? max : c);

    float min = (a < b ? a : b);
    min = (min < c ? min : c);

    cout << "Max = " << max << endl;
    cout << "Min = " << min << endl;
    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    float a[100];
    int duong = 0, am = 0, bang0 = 0;

    for (int i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] > 0) duong++;
        else if (a[i] < 0) am++;
        else bang0++;
    }

    cout << "Duong: " << duong << endl;
    cout << "Am: " << am << endl;
    cout << "Bang 0: " << bang0 << endl;
    return 0;
}
