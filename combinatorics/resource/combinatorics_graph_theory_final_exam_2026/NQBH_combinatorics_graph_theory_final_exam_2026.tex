	\item(b) \emph{(10 điểm)} Giải câu (a) với $n\in\mathbb{N}^\star$ tổng quát. Viết cụ thể $V,E,|V|,|E|$ \& tính $\deg v$ với mỗi $v\in V$.
\end{baitoan}
